# 查看当前状态
  git status

# 把文件添加到跟踪中
  git add -A

# 提交到本地
  git commit
  git commit -m "注释"

# 推送到远程服务器
  git push

# 撤销git add添加的所有改动
  git reset HEAD

# 从远程服务器下载
  git pull

# 查看远程服务器地址
  git remote -v

# 查看分支
  git branch

# 删除一个文件
  git rm 文件

# 重命名文件夹
  git mv 源文件夹 目标文件夹
